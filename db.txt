CREATE TABLE admin_users (
    id SERIAL PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    role VARCHAR(20) DEFAULT 'staff',
    real_name VARCHAR(50),
    status VARCHAR(20) DEFAULT 'active',
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE farm_users (
    id SERIAL PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    gender VARCHAR(10),                          -- male/female
    phone_number VARCHAR(20) UNIQUE NOT NULL,
    wechat_number VARCHAR(50),
    real_name VARCHAR(50),
    status VARCHAR(20) DEFAULT 'active',         -- active/disabled
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE customer_users (
    id SERIAL PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    gender VARCHAR(10),
    phone_number VARCHAR(20) UNIQUE NOT NULL,
    wechat_number VARCHAR(50),
    real_name VARCHAR(50),
    region VARCHAR(100),                         -- 地区，如"北京市朝阳区"
    company VARCHAR(100),                        -- 公司/学校
    status VARCHAR(20) DEFAULT 'active',
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE favorites (
    id SERIAL PRIMARY KEY,
    customer_id INTEGER REFERENCES customer_users(id),  -- 关联游客用户
    type VARCHAR(20) NOT NULL,                   -- farm / activity
    target_id INTEGER NOT NULL,                  -- farms.id 或 activities.id
    created_at TIMESTAMP DEFAULT NOW(),
    UNIQUE(customer_id, type, target_id)
);

CREATE TABLE farms (
    id SERIAL PRIMARY KEY,
    owner_id INTEGER REFERENCES farm_users(id),  -- 关联农场主用户表
    name VARCHAR(100) NOT NULL,
    location VARCHAR(200),
    description TEXT,
    features TEXT,
    pain_points TEXT,
    status VARCHAR(20) DEFAULT 'pending',
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE ai_plans (
    id SERIAL PRIMARY KEY,
    farm_id INTEGER REFERENCES farms(id),
    plan_name VARCHAR(200) NOT NULL,
    days INTEGER DEFAULT 1,
    has_meal BOOLEAN DEFAULT FALSE,
    has_stay BOOLEAN DEFAULT FALSE,              -- 是否住宿（新增）
    user_input TEXT,
    ai_output TEXT,
    cost_estimate TEXT,
    token_usage INTEGER,                         -- AI调用消耗的token数
    token_cost DECIMAL(10,4),                    -- 预估费用（元）
    status VARCHAR(20) DEFAULT 'pending',
    published_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE activities (
    id SERIAL PRIMARY KEY,
    farm_id INTEGER REFERENCES farms(id),
    plan_id INTEGER REFERENCES ai_plans(id),
    title VARCHAR(200) NOT NULL,
    description TEXT,
    days INTEGER DEFAULT 1,                      -- 天数
    has_meal BOOLEAN DEFAULT FALSE,              -- 是否含餐
    has_stay BOOLEAN DEFAULT FALSE,              -- 是否住宿
    status VARCHAR(20) DEFAULT 'draft',          -- draft/published/sold_out/cancelled
    published_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE activity_prices (
    id SERIAL PRIMARY KEY,
    activity_id INTEGER REFERENCES activities(id),
    price DECIMAL(10,2) NOT NULL,
    option_name VARCHAR(50) NOT NULL,            -- 选项名称，如"成人"、"儿童"、"家庭套票"
    option_description TEXT,                     -- 选项描述，如"12岁以上"
    sort_order INTEGER DEFAULT 0,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE b_orders (
    id SERIAL PRIMARY KEY,
    order_no VARCHAR(50) UNIQUE NOT NULL,
    owner_id INTEGER REFERENCES farm_users(id),   -- 关联农场主用户
    farm_id INTEGER REFERENCES farms(id),
    order_type VARCHAR(20),                       -- plan_fee / monthly_fee
    amount DECIMAL(10,2) NOT NULL,
    payment_status VARCHAR(20) DEFAULT 'pending',
    payment_time TIMESTAMP,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE c_orders (
    id SERIAL PRIMARY KEY,
    order_no VARCHAR(50) UNIQUE NOT NULL,
    activity_id INTEGER REFERENCES activities(id),
    farm_id INTEGER REFERENCES farms(id),
    owner_id INTEGER REFERENCES farm_users(id),   -- 关联农场主用户
    customer_id INTEGER REFERENCES customer_users(id),  -- 关联游客用户
    quantity INTEGER DEFAULT 1,                   -- 总人数（替代 adult_count + child_count）
    total_amount DECIMAL(10,2) NOT NULL,
    platform_commission DECIMAL(10,2),
    order_status VARCHAR(20) DEFAULT 'pending',
    payment_time TIMESTAMP,
    created_at TIMESTAMP DEFAULT NOW()
);
